cd employees_db
mysql -t < employees.sql
wget http://downloads.mysql.com/docs/sakila-db.tar.gz
tar xzvf sakila-db.tar.gz && rm -f sakila-db.tar.gz
mysql < sakila-db/sakila-schema.sql
mysql sakila < sakila-db/sakila-data.sql
